Basic Replication Setup

-- Configure Extract process
ADD EXTRACT ext1, TRANLOG, BEGIN NOW
ADD EXTTRAIL ./dirdat/aa, EXTRACT ext1

-- Configure Replicat process
ADD REPLICAT rep1, EXTTRAIL ./dirdat/aa
START EXTRACT ext1
START REPLICAT rep1

Vector Replication for AI

-- Replicate vector data to AI applications
CREATE TABLE ai_vectors (
    id NUMBER PRIMARY KEY,
    content CLOB,
    embedding VECTOR(384)
);

ADD EXTRACT ai_ext, TRANLOG, BEGIN NOW
ADD EXTTRAIL ./dirdat/ai, EXTRACT ai_ext
TABLE schema.ai_vectors;

ADD REPLICAT ai_rep, EXTTRAIL ./dirdat/ai
MAP schema.ai_vectors, TARGET ai_vectors;
